  <action>
    Create the scripts/audit-licenses.js file that:
    
    1. Reads src-tauri/Cargo.lock to extract Rust crate names
    2. Reads package-lock.json to extract npm package names  
    3. For each dependency, attempts to determine license information
    4. Counts total packages and copyleft-licensed packages
    5. Writes result to src/generated/license-audit.json
    
    Implementation approach:
    - For npm packages: use `npm ls --json` or parse package-lock.json and check node_modules/{pkg}/package.json for license field
    - For Rust crates: use `cargo license` if available, or parse Cargo.lock and fetch license info from crates.io API (or use a simpler heuristic based on known copyleft crates)
    
    Since this is build-time only and simpler is better:
    - Parse package-lock.json for npm package names
    - Parse Cargo.lock for Rust crate names
    - Use a conservative approach: check package.json files in node_modules for SPDX license identifiers
    - Match against known copyleft license SPDX identifiers
    
    Script structure:
    ```javascript
    #!/usr/bin/env node
    const fs = require('fs');
    const path = require('path');

    const COPYLEFT_LICENSES = [
      'GPL-2.0', 'GPL-3.0', 'GPL-2.0+', 'GPL-3.0+',
      'AGPL-1.0', 'AGPL-3.0',
      'LGPL-2.0', 'LGPL-2.1', 'LGPL-3.0',
      'CC-BY-SA-2.0', 'CC-BY-SA-3.0', 'CC-BY-SA-4.0',
      'MPL-1.0', 'MPL-1.1', 'MPL-2.0',
      'CDDL-1.0', 'CDDL-1.1',
      'EPL-1.0', 'EPL-2.0',
      'CPL-1.0'
    ];

    function isCopyleftLicense(license) {
      if (!license) return false;
      const normalized = license.toString().toUpperCase();
      return COPYLEFT_LICENSES.some(cl => normalized.includes(cl));
    }

    function parseCargoLock() {
      // Parse Cargo.lock TOML-like format
      // Return array of { name, version, license }
    }

    function parsePackageLock() {
      // Parse package-lock.json
      // Return array of { name, version, license }
    }

    function auditLicenses() {
      const npmPackages = parsePackageLock();
      const rustCrates = parseCargoLock();
      
      const allDeps = [...npmPackages, ...rustCrates];
      const totalPackages = allDeps.length;
      const copyleftCount = allDeps.filter(d => isCopyleftLicense(d.license)).length;
      
      const result = {
        totalPackages,
        copyleftCount,
        generatedAt: new Date().toISOString()
      };
      
      // Ensure directory exists
      const outputDir = path.join(__dirname, '..', 'src', 'generated');
      if (!fs.existsSync(outputDir)) {
        fs.mkdirSync(outputDir, { recursive: true });
      }
      
      fs.writeFileSync(
        path.join(outputDir, 'license-audit.json'),
        JSON.stringify(result, null, 2)
      );
      
      console.log(`License audit complete: ${totalPackages} packages, ${copyleftCount} copyleft`);
    }

    auditLicenses();
    ```
    
    For npm license detection:
    - Iterate through package-lock.json packages
    - For each package, try to read node_modules/{name}/package.json
    - Extract the "license" field (can be string or object like { type: "MIT" })
    
    For Rust license detection (simpler approach):
    - Try to use `cargo license --json` if cargo-license is installed
    - Fallback: just count crates from Cargo.lock without license info (conservative: assume permissive)
    - Or install cargo-license as part of build process
